Spring Boot With Docker Note

Dockerfile

通过java启动参数来设置时区

FROM frolvlad/alpine-oraclejdk8:slim
MAINTAINER eastseven , <eastseven@foxmail.com>

VOLUME /logs
VOLUME /upload
EXPOSE 8080

ADD app.jar app.jar
RUN sh -c 'touch app.jar'
ENV JAVA_OPTS="-Xmx300m -Xms300m"
ENTRYPOINT [ "sh", "-c", "java $JAVA_OPTS -Djava.security.egd=file:/dev/./urandom -jar -Duser.timezone=GMT+8 app.jar" ]

build

docker build -t eastseven/spring-boot .

run

docker run --rm -p 8080:8080 -v $(pwd)/logs:/logs -v $(pwd)/upload:/upload eastseven/spring-boot
